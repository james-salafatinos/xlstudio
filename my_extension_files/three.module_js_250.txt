e 'DirectionalLight':
					uniforms = {
						shadowBias: 0,

					};
					break;

				case 'SpotLight':

						shadowMapSize: new Vector2()
					};
					break;

						shadowBias: 0,
						shadowNormalBias: 0,
						shadowRadius: 1,
						shadowMapSize: new Vector2(),
						shadowCameraNear: 1,
						shadowCameraFar: 1000
					};
					break;

				// TODO (abelnation): set RectAreaLight shadow uniforms

			}

			lights[ light.id ] = uniforms;

			return uniforms;

		}

	};

}
